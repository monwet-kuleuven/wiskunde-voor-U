\begin{abstract}
	Bekijk alles positief!

\maketitle

\begin{definition}
	Voor een reëel getal $a\in\R$ definiëren we de \textit{absolute waarde} van $a$, genoteerd $|a|$, als
	\[
		|a| \overset{def}{=}\displaystyle\ 
		          \left\{
			\begin{array}{rll  } 
				a  & \mbox{als} & a \geqslant 0, \\
				-a & \mbox{als} & a<0.
			\end{array}\right.
	\]
\end{definition}
